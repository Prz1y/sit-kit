==========================================================================

==========================================================================
 第 2 步: 网卡调优 [两台机器都执行, 改成自己的网卡名]
 MTU 9000 jumbo frame; 7 队列匹配 7 IRQ CPU; ring 8192 防溢出;
 固定中断合并 128μs; 全开 offload; RSS 均匀分流;
 清除 RPS 避免与硬件 RSS 冲突(实测冲突掉 40G);
 conntrack NOTRACK 绕过每包连接跟踪开销
==========================================================================

NIC1=ens2f0np0
NIC2=ens2f1np1

for NIC in $NIC1 $NIC2; do
  ip link set dev $NIC mtu 9000
  ethtool -L $NIC combined 7
  ethtool -G $NIC rx 8192 tx 8192
  ethtool -C $NIC adaptive-rx off adaptive-tx off rx-usecs 128 tx-usecs 128
  ethtool -K $NIC rx on tx on tso on gso on gro on lro on
  ethtool -K $NIC tx-nocache-copy off
  ethtool -K $NIC ntuple on rxhash on
  ethtool -A $NIC rx off tx off
  ethtool -X $NIC equal 7
  ethtool -N $NIC rx-flow-hash tcp4 sdfn
  ethtool -N $NIC rx-flow-hash udp4 sdfn
done

for NIC in $NIC1 $NIC2; do
  for f in /sys/class/net/$NIC/queues/rx-*/rps_cpus; do
    echo 0 > $f 2>/dev/null
  done
  echo "已清除 $NIC 所有 RPS"
done

iptables -t raw -F
iptables-restore <<'EOF'
*raw
:PREROUTING ACCEPT [0:0]
:OUTPUT ACCEPT [0:0]
-A PREROUTING -p tcp --dport 5001:5100 -j NOTRACK
-A PREROUTING -p tcp --dport 6001:6100 -j NOTRACK
-A PREROUTING -p tcp --sport 5001:5100 -j NOTRACK
-A PREROUTING -p tcp --sport 6001:6100 -j NOTRACK
-A OUTPUT     -p tcp --sport 5001:5100 -j NOTRACK
-A OUTPUT     -p tcp --sport 6001:6100 -j NOTRACK
COMMIT
EOF


==========================================================================
 第 3 步: IRQ 绑核 [两台机器都执行]
 将 NIC1/NIC2 的 mlx5 中断向量分别绑到网卡所在 NUMA 的
 前 7 / 后 7 个 CPU, 1:1 对应 7 个队列, 避免跨 NUMA 处理中断
==========================================================================

cat /sys/class/net/$NIC1/device/numa_node

numactl -H | grep "node 0 cpus:"

numactl -H | grep "node 5 cpus:"

CPUS_NIC1=(0 1 2 3 4 5 6)

PCI=$(basename $(readlink -f /sys/class/net/$NIC1/device))
i=0
for irq in $(grep "mlx5_comp.*$PCI" /proc/interrupts | awk -F: '{print $1}' | tr -d ' '); do
  echo ${CPUS_NIC1[$((i % 7))]} > /proc/irq/$irq/smp_affinity_list
  i=$((i+1))
done
echo "NIC1 ($NIC1) IRQ 绑核完成: $i 个中断向量"

CPUS_NIC2=(7 8 9 10 11 12 13)

PCI=$(basename $(readlink -f /sys/class/net/$NIC2/device))
i=0
for irq in $(grep "mlx5_comp.*$PCI" /proc/interrupts | awk -F: '{print $1}' | tr -d ' '); do
  echo ${CPUS_NIC2[$((i % 7))]} > /proc/irq/$irq/smp_affinity_list
  i=$((i+1))
done
echo "NIC2 ($NIC2) IRQ 绑核完成: $i 个中断向量"
